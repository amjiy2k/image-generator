// DOM Elements const form = document.getElementById('imageGeneratorForm'); const promptInput = document.getElementById('prompt'); const styleSelect = document.getElementById('style'); const qualitySelect = document.getElementById('quality'); const resultsSection = document.getElementById('results'); const generatedImage = document.getElementById('generatedImage'); const imagePlaceholder = document.querySelector('.image-placeholder'); const downloadBtn = document.getElementById('downloadBtn'); const loadingIndicator = document.getElementById('loading'); const toast = document.getElementById('toast'); // Event Listeners form.addEventListener('submit', generateImage); downloadBtn.addEventListener('click', downloadImage); // Generate Image Function async function generateImage(e) { e.preventDefault(); // Show loading indicator loadingIndicator.style.display = 'flex'; imagePlaceholder.style.display = 'none'; generatedImage.style.display = 'none'; // Prepare the prompt let prompt = promptInput.value.trim(); const style = styleSelect.value; const quality = qualitySelect.value; if (style) { prompt += `, ${style} style`; } if (!prompt) { showToast('Please enter a prompt', 'error'); loadingIndicator.style.display = 'none'; imagePlaceholder.style.display = 'flex'; return; } try { // Call Hugging Face API const response = await fetch( "https://api-inference.huggingface.co/models/runwayml/stable-diffusion-v1-5", { method: "POST", headers: { "Authorization": `Bearer ${HF_API_TOKEN}`, "Content-Type": "application/json" }, body: JSON.stringify({ inputs: prompt, options: { wait_for_model: true } }) } ); if (!response.ok) { throw new Error(`API request failed with status ${response.status}`); } // Get the image blob const imageBlob = await response.blob(); const imageUrl = URL.createObjectURL(imageBlob); // Display the generated image generatedImage.src = imageUrl; generatedImage.style.display = 'block'; loadingIndicator.style.display = 'none'; resultsSection.style.display = 'block'; // Enable download button downloadBtn.dataset.imageUrl = imageUrl; // Show success message showToast('Image generated successfully!', 'success'); // Scroll to results resultsSection.scrollIntoView({ behavior: 'smooth' }); } catch (error) { console.error('Error generating image:', error); loadingIndicator.style.display = 'none'; imagePlaceholder.style.display = 'flex'; showToast('Failed to generate image. Please try again.', 'error'); } } // Download Image Function function downloadImage() { const imageUrl = downloadBtn.dataset.imageUrl; if (!imageUrl) return; const link = document.createElement('a'); link.href = imageUrl; link.download = `ai-generated-${Date.now()}.jpg`; document.body.appendChild(link); link.click(); document.body.removeChild(link); showToast('Download started!', 'success'); } // Show Toast Notification function showToast(message, type) { toast.innerHTML = ` ${message} `; toast.className = `toast ${type} show`; setTimeout(() => { toast.classList.remove('show'); }, 3000); } // Mobile Menu Toggle const mobileMenuBtn = document.querySelector('.mobile-menu'); const navLinks = document.querySelector('.nav-links'); mobileMenuBtn.addEventListener('click', () => { navLinks.style.display = navLinks.style.display === 'flex' ? 'none' : 'flex'; }); // Close mobile menu when clicking on a link document.querySelectorAll('.nav-links a').forEach(link => { link.addEventListener('click', () => { if (window.innerWidth <= 768) { navLinks.style.display = 'none'; } }); }); // Responsive adjustments window.addEventListener('resize', () => { if (window.innerWidth > 768) { navLinks.style.display = 'flex'; } else { navLinks.style.display = 'none'; } });
